useDebounce.ts 
export function useDebounce<T>(value: T, ms = 300) {
  const [v, setV] = useState(value)
  useEffect(() => {
    const id = setTimeout(() => setV(value), ms)
    return () => clearTimeout(id)   // cleanup on change
  }, [value, ms])
  return v
}
